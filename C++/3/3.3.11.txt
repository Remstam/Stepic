#include <cstddef> // size_t
#include <cstring> // strlen, strcpy

struct String {

    /* Реализуйте этот конструктор */
	String(size_t n, char c)
    {
        size = n;
        str = new char[n + 1];
        
        char* s = str;
        for (int i = 0; i < n; i++)
            *s++ = c;
        *s = '\0';
    }

    /* и деструктор */
	~String()
    {
        delete[] str;
    }

	size_t size;
	char *str;
};
